25fn parse_fen(fen: &str) -> Board {
26    let fen_vec = fen.split(' ').collect::<Vec<&str>>();
27    //Split the FEN into its constituent parts
28
29    let board_state_vec = fen_vec[0].split('/').collect::<Vec<&str>>();
30    let mut row = vec![];
31    let mut board_state = vec![];
32    for single_row in board_state_vec {
33        for character in single_row.chars() {
34            //assuming valid FEN (only characters and numbers)
35            const RADIX: u32 = 10;
36            if character.is_numeric() {
37                for _i in 0..character
38                    .to_digit(RADIX)
39                    .expect("Could not convert char to int")
40                {
41                    row.push('*');
42                }
43            } else {
44                row.push(character);
45            }
46        }
47        board_state.push(row.clone());
48        row.retain(|_x| false); // empty the vector
49    }
50    //Parse the board state part of the FEN into a 2d nested Vec
51
52    Board {
53        board_state,
54        active_player: fen_vec[1].chars().next().expect("string is empty"),
55        castling_availability: fen_vec[2].to_string(),
56        en_passant_square: fen_vec[3].to_string(),
57        halfmove_counter: fen_vec[4]
58            .parse::<i32>()
59            .expect("I'm afraid it is not possible to convert this value to i32, my good sir."),
60        turn_counter: fen_vec[5]
61            .parse::<u64>()
62            .expect("I'm afraid it is not possible to convert this value to u64, my good sir."),
63        promotion_selection: 'q',
64    }
65    //Then feed the rest directly into the cosntructor
66} // Creates a Board struct from any given FEN. Inverse function to generate_fen()

68fn generate_fen(board: &Board) -> String {
69    let mut fen = String::new();
70    let mut fen_row = String::new();
71    let mut empty_squares: i32 = 0;
72    for row in board.board_state.clone() {
73        for char in row {
74            if char == '*' {
75                empty_squares += 1;
76            } else if empty_squares > 0 {
77                fen_row = format!("{}{}{}", fen_row, empty_squares, char);
78                empty_squares = 0;
79            } else {
80                fen_row = format!("{}{}", fen_row, char);
81            }
82        }
83        if empty_squares > 0 {
84            fen_row = format!("{}{}", fen_row, empty_squares);
85            empty_squares = 0;
86        } // In case the last few squares are empty, add them
87        if fen.is_empty() {
88            fen = fen_row.clone();
89        } else {
90            fen = format!("{}/{}", fen, fen_row);
91        }
92        fen_row = "".to_string();
93    }
94    format!(
95        "{} {} {} {} {} {}",
96        fen,
97        board.active_player,
98        board.castling_availability,
99        board.en_passant_square,
100        board.halfmove_counter,
101        board.turn_counter
102    )
103} // Creates a FEN from any given Board struct. Inverse function for parse_fen().

105fn get_board_coords(algebraic_notation: String) -> Vec<i32> {
106    let col_names = "abcdefgh".to_string();
107    let col_number_index = col_names
108        .chars()
109        .position(|y| y == algebraic_notation.chars().nth(0).unwrap())
110        .expect("Unable to find col number");
111
112    let col_number = i32::try_from(col_number_index).expect("Column number index too large");
113
114    let row_number_temp = algebraic_notation.chars().nth(1).unwrap().to_digit(10).unwrap();
115    let row_number = i32::try_from(row_number_temp).ok().unwrap();
116    vec![8 - row_number, col_number]
117} // Generates Board.game_state coords from algebraic notation. [0~7, 0~7]. [0, 0] corresponds to a8, and [7,7] is h1. [3,4] is e5.

119fn get_algebraic_notation(coords: Vec<i32>) -> String {
120    let col_names = "abcdefgh".to_string();
121    let col_name = col_names.chars().nth(coords[1] as usize).expect("Blimey! Unable to find this col!");
122    format!("{}{}", col_name, 8 - coords[0])
123} // Generates algebraic notation from Board.game_state coords. Inverse to get_board_coords.

138fn check_for_checks(board: &Board) -> Vec<bool> {
139    vec![player_is_in_check(board, 'w'), player_is_in_check(board, 'b')]
140}
141fn player_is_in_check(board: &Board, player: char) -> bool {
142    let mut enemy = 'n';
143    if player == 'w' {
144        enemy = 'b';
145    } else if player == 'b' {
146        enemy = 'w';
147    }
148    // For each move in the enemy pieces, check for a threatened king. If such is the case, return true.
149    for (_key, value) in get_available_moves_internal(board.clone(), enemy, true) {
150        for legal_move in value {
151            // println!("Legal move for {} at {:?}: {}", board.board_state[_key[0] as usize][_key[1] as usize], _key, board.board_state[legal_move[0] as usize][legal_move[1] as usize]);
152            if board.board_state[legal_move[0] as usize][legal_move[1] as usize].eq_ignore_ascii_case(&'k') {
153                return true;
154            }
155        }
156    }
157    false // Check if any opposing piece threatens the king, if yes, return true, else return false
158} // Returns true if the player is in check

165/// A function to return available moves for a given color on a given board. 
166///
167/// ## Arguments
168/// ```
169/// mut board: Board, // The Board to look at. Usually your_game.board.
170/// color: char, // the color to return moves for. usually your_game.board.active_player. 
171/// force_no_check: bool // Whether or not to remove moves that would not take the color's king out of check.
172/// 
173/// ```
174/// ## Returns
175/// This function returns a HashMap, where
176/// key: piece location,
177/// value: each square that the piece can move to.
178///
179/// ## Example
180///
181/// ```
182/// let example_game = Game::new(); // Create a new game at the starting position
183/// let moves = get_available_moves(example_game.board, example_game.board.active_player, false)
184/// assert_eq!(moves["e2".to_string()], ["e3", "e4"]) // Available moves for the e2 pawn
185/// ```
186pub fn get_available_moves(mut board: Board,
187    color: char,
188    force_no_check: bool) -> HashMap<String, Vec<String> > {
189        let mut temp_keys: Vec<String> = vec![];
190        let mut temp_values: Vec<Vec<String>> = vec![];
191        let mut new_hashmap: HashMap<String, Vec<String> > = HashMap::new();
192        for (key, value) in get_available_moves_internal(board, color, force_no_check).iter_mut() {
193            temp_keys.push(get_algebraic_notation(key.clone()));
194            temp_values.push(value.iter().map(|x| get_algebraic_notation(x.clone()) ).collect());
195        }
196        for (index, value) in temp_keys.iter().enumerate() {
197            new_hashmap.insert(value.clone(), temp_values[index].clone());
198        }
199        new_hashmap
200    }

203fn get_available_moves_internal (
204    mut board: Board,
205    color: char,
206    force_no_check: bool
207) -> HashMap< Vec<i32>, Vec<Vec<i32>> > {
208    let mut output = HashMap::new();
209    for (y_pos, row) in board.clone().board_state.iter().enumerate() {
210        for (x_pos, piece) in row.iter().enumerate() {
211            if color == 'w' && piece.is_ascii_uppercase() {
212                // WHITE pieces are represented by UPPERCASE letters
213                let coords: Vec<i32> = vec![i32::try_from(y_pos).unwrap(), i32::try_from(x_pos).unwrap()];
214                let movements = board.get_piece_movements(&coords, &piece, &color);
215                if !movements.is_empty() {
216                    output.insert(coords, movements);
217                }
218                    
219            } else if color == 'b' && piece.is_ascii_lowercase() {
220                // black pieces are represented by lowercase letters
221                let coords: Vec<i32> = vec![i32::try_from(y_pos).unwrap(), i32::try_from(x_pos).unwrap()];
222                let movements = board.get_piece_movements(&coords, &piece, &color);
223                if !movements.is_empty() {
224                    output.insert(coords, movements);
225                }
226                    
227            }
228        }
229    }
230    if !force_no_check {//DO NOT RUN THIS IF CALLED BY FN PLAYER IS IN CHECK
231        if player_is_in_check(&board, color) {
232            for (key, value) in output.iter_mut() {
233                value.retain(|legal_move| -> bool { // retain all moves where x isnt in check
234                    let mut test_board = board.clone();
235                    test_board.move_piece(key.clone(), legal_move.to_vec());
236                    !player_is_in_check(&test_board, color)
237                });
238            }
239            //Remove castling moves if player is in check
240            if color == 'w' {
241                if board.board_state[7][4] == 'K' {
242                    if output[&vec![7,4]].contains(&vec![7,6]) {
243                        let index = output[&vec![7,4]].iter().position(|x| *x == vec![7,6]).unwrap();
244                        output.get_mut(&vec![7,4]).unwrap().remove(index);
245                    }
246                    if output[&vec![7,4]].contains(&vec![7,2]) {
247                        let index = output[&vec![7,4]].iter().position(|x| *x == vec![7,2]).unwrap();
248                        output.get_mut(&vec![7,4]).unwrap().remove(index);
249                    }
250                }
251            } else if color == 'b' {
252                if board.board_state[0][4] == 'k' {
253                    if output[&vec![0,4]].contains(&vec![0,6]) {
254                        let index = output[&vec![0,4]].iter().position(|x| *x == vec![0,6]).unwrap();
255                        output.get_mut(&vec![0,4]).unwrap().remove(index);
256                    }
257                    if output[&vec![0,4]].contains(&vec![0,2]) {
258                        let index = output[&vec![0,4]].iter().position(|x| *x == vec![0,2]).unwrap();
259                        output.get_mut(&vec![0,4]).unwrap().remove(index);
260                    }
261                }
262            }
263        }
264        let mut to_remove = Vec::new();   
265        for (key, value) in &output {
266            if value.is_empty() {
267                to_remove.push(key.to_owned());
268            }
269        }
270        for key in to_remove.iter() {
271            output.remove(key);
272        }
273    } // println!("{:?}", output);
274    output
275} // For any given color, finds pieces of that color. Returns a 
276//Hashmap of coords with pieces of that color, and available moves for each coordinate.

284/// A struct to represent the chess game itself.
285///
286/// ## Attributes
287/// ```
288/// fen: String, // the FEN string that represents the current position.
289/// board: Board, // A representation of the FEN string that is easier to work with.
290/// checks: Vec<bool>, // index 0 is white's check status, index 1 is black's check status
291/// game_status: u8, // 0: Game in progress, 1: Checkmate (White wins), 2: Checkmate (Black wins), 3: Stalemate, 4: Draw by 50 move rule
292/// 
293/// ```
294pub struct Game {
295    fen: String,
296    board: Board, 
297    checks: Vec<bool>, // index 0 is white's check status, index 1 is black's check status
298    game_status: u8, // 0: Game in progress, 1: Checkmate (White wins), 2: Checkmate (Black wins), 3: Stalemate, 4: Draw by 50 move rule
299    
300}

306    /// fen: String, // A valid FEN string.

308    /// ```
309    /// ## Returns

311    ///
312    /// ## Example
313    ///
314    /// ```
315    /// let example_game = Game::new_from_fen("r1bk3r/p2pBpNp/n4n2/1p1NP2P/6P1/3P4/P1P1K3/q5b1 b - - 1 23".to_string()); // Create a new game at the final position of the Immortal Game. This Game will have status 1, since White is victorious.

317    pub fn new_from_fen(fen: String) -> Game {

327    /// ```
328    ///
329    /// ## Returns

342    /// A function to make a move in the Game. Automatically detects whose turn it is based on `Game.board`.
343    ///
344    /// ## Arguments
345    /// ```
346    /// source: String, // The square where the piece to move stands, in algebraic notation.
347    /// target: String, // The square to which to move the piece, in algebraic notation.
348    /// 
349    /// ```
350    /// ## Returns
351    /// This function returns true if the move was successfully made, false otherwist,
352    ///
353    /// ## Example
354    ///
355    /// ```
356    /// let example_game = Game::new(); // Create a new game at the starting position
357    /// example_game.make_move("f2".to_string(), "f3".to_string()); // 1. f3
358    /// example_game.make_move("e7".to_string(), "e5".to_string()); // 1... e5
359    /// example_game.make_move("g2".to_string(), "g4".to_string()); // 2. g4
360    /// example_game.make_move("d8".to_string(), "h4".to_string()); // 2... Qh4#
361    /// ```
362    pub fn make_move(&mut self, source: String, target: String) -> bool { //Returns true if a valid move has been made
363        //Assuming both square and target are valid algebraic notation.
364        let source_coords = get_board_coords(source);
365        let target_coords = get_board_coords(target);
366        let available_moves =
367            get_available_moves_internal(self.board.clone(), self.board.active_player, false);
368        if available_moves.contains_key(&source_coords) &&
369             available_moves[&source_coords].contains(&target_coords) {
370                // hopefully error free way of checking if the move is a valid move as dictated by get_available_moves_internal()
371                println!("Source coords: {:?}, Target coords: {:?}", &source_coords, &target_coords);
372                self.board.move_piece(source_coords, target_coords);
373                //self.fen = generate_fen(self.board.clone());
374        } else {
375            return false;
376        }
377
378        if self.board.active_player == 'w' {
379            self.board.active_player = 'b';
380        } else if self.board.active_player == 'b' {
381            self.board.turn_counter += 1;
382            self.board.active_player = 'w';
383        }
384
385        self.checks = check_for_checks(&self.board);
386
387        self.update_game_status();
388
389        self.fen = generate_fen(&self.board);
390        
391        true
392    } // TODO Make move if move is available for the active player, then switch active player, then check for checks
393
394    fn update_game_status(&mut self) {
395        self.game_status = 0;
396        if self.board.halfmove_counter >= 100 {
397            self.game_status = 4;
398        }
399        //check for checkmate
400        if get_available_moves_internal(self.board.clone(), 'w', false).keys().len() == 0 {
401            if self.checks[0] {
402                self.game_status = 2;
403                return;
404            } else {
405                self.game_status = 3;
406                return;
407            }
408        }
409        if get_available_moves_internal(self.board.clone(), 'b', false)
410        .keys()
411        .len() == 0 {
412            if self.checks[1] {
413                self.game_status = 1;
414                return;
415            } else {
416                self.game_status = 3;
417                return;
418            }
419        }
420    }

438/// A struct to represent the chessboard.
439///
440/// ## Attributes
441/// ```
442/// board_state: Vec<Vec<char>>,
443/// // Represents the board. Pieces are represented by their FEN notation (capital for white, lowercase for black)
444/// // Blank squares are represented by "*"
445/// active_player: char, // 'w' or 'b'. Will produce undocumented behaviour if set to anything else.
446/// 
447/// castling_availability: String,
448/// //This value is in the power set of string "KQkq" and
449/// // represents which castling moves are available. When no castling moves are available, the value will be "-".
450/// // Castling not implemented yet.
451/// en_passant_square: String,
452/// // This value represents whether or not en passant is available,
453/// // and if so, the square to which the capturing pawn will move. Otherwise,
454/// // the value will be "-".
455/// // En passant not implemented yet.
456/// halfmove_counter: i32,
457/// // This counter increments for every move made without a capture
458/// // or a pawn move. Otherwise, it resets.
459///  // When it reaches 100, the game is a draw.
460/// turn_counter: u64,
461/// // This counter increments by one every time Black makes a move.
462/// promotion_selection: char,
463/// // Selected piece that a pawn promotes to. Case-insensitive. This selection applies to both Defaults to 'q' on each parse_fen call.
464/// ```
465pub struct Board {
466    board_state: Vec<Vec<char>>,
467    // Represents the board. Pieces are represented by their FEN notation (capital for white, lowercase for black)
468    // Blank squares are represented by "*"
469    active_player: char, // "w" or "b"
470    castling_availability: String,
471    //This value is in the power set of string "KQkq" and
472    // represents which castling moves are available. When no castling moves are available, the value will be "-".
473    // Castling not implemented yet.
474    en_passant_square: String,
475    // This value represents whether or not en passant is available,
476    // and if so, the square to which the capturing pawn will move. Otherwise,
477    // the value will be "-".
478    // En passant not implemented yet.
479    halfmove_counter: i32,
480    // This counter increments for every move made without a capture
481    // or a pawn move. Otherwise, it resets.
482    // When it reaches 100, the game is a draw.
483    turn_counter: u64,
484    // This counter increments by one every time Black makes a move.
485    promotion_selection: char,
486    // Selected piece that a pawn promotes to. Defaults to q on each parse_fen call.
487}
488impl Board {
489    fn get_piece_movements(&mut self, coords: &Vec<i32>, piece: &char, color: &char) -> Vec<Vec<i32>> {
490        
491        let mut move_list = vec![];
492        let x_pos = coords[1];
493        let y_pos = coords[0];
494        //println!("Matching piece movements: {}", piece.to_ascii_lowercase());
495                // println!("{:?}", board);
496        match piece.to_ascii_lowercase() {
497            'p' => {
498                if color == &'w' {
499                    // White pawns move in -y
500                    if self.board_state[(y_pos-1) as usize][(x_pos) as usize] == '*' {
501                        move_list.push(vec![y_pos-1, x_pos]);
502                    }   
503                    // pawns can take diagonally.
504                    if x_pos - 1 > 0 {
505                        if is_enemy_piece('w', self.board_state[(y_pos-1) as usize][(x_pos-1) as usize]) {
506                            move_list.push(vec![y_pos-1, x_pos-1]);
507                        }
508                    }
509                    if x_pos + 1 < 8 {
510                        if is_enemy_piece('w', self.board_state[(y_pos-1) as usize][(x_pos+1) as usize]) {
511                            move_list.push(vec![y_pos-1, x_pos+1]);
512                        }
513                    }
514                    //Pawn First Move Advance
515                    if (y_pos == 6) && self.board_state[(y_pos-2) as usize][(x_pos) as usize] == '*' && self.board_state[(y_pos-1) as usize][(x_pos) as usize] == '*'{
516                        move_list.push(vec![y_pos-2, x_pos]);
517                    }
518                    if self.en_passant_square != "-" { // en passant is available
519                        if x_pos - 1 > 0 {
520                            if (get_board_coords(self.en_passant_square.clone()) == vec![y_pos-1, x_pos-1]) {
521                                move_list.push(vec![y_pos-1, x_pos-1]);
522                            }
523                        }
524                        if x_pos + 1 < 8 {
525                            if (get_board_coords(self.en_passant_square.clone()) == vec![y_pos-1, x_pos+1]) {
526                                move_list.push(vec![y_pos-1, x_pos+1]);
527                            }
528                        }
529                    }
530                } else if color == &'b' {
531                    // black pawns move in +y
532                    if self.board_state[(y_pos+1) as usize][(x_pos) as usize] == '*' {
533                        move_list.push(vec![y_pos+1, x_pos]);
534                    }
535                    // pawns can take diagonally.
536                    if x_pos - 1 > 0 {
537                        if is_enemy_piece('b', self.board_state[(y_pos+1) as usize][(x_pos-1) as usize]) {
538                            move_list.push(vec![y_pos+1, x_pos-1]);
539                        }
540                    }
541                    if x_pos + 1 < 8 {
542                        if is_enemy_piece('b', self.board_state[(y_pos+1) as usize][(x_pos+1) as usize]) {
543                            move_list.push(vec![y_pos+1, x_pos+1]);
544                        }
545                    }
546                    //Pawn First Move Advance
547                    if (y_pos == 1) && self.board_state[(y_pos+2) as usize][(x_pos) as usize] == '*' && self.board_state[(y_pos+1) as usize][(x_pos) as usize] == '*' {
548                        move_list.push(vec![y_pos+2, x_pos]);
549                    }
550                    if self.en_passant_square != "-" { // en passant is available
551                        if x_pos - 1 > 0 {
552                            if (get_board_coords(self.en_passant_square.clone()) == vec![y_pos+1, x_pos-1]) {
553                                move_list.push(vec![y_pos+1, x_pos-1]);
554                            }

907    // giant match-case statement which returns a set of moves for each piece
908    fn move_piece(&mut self, source: Vec<i32>, target: Vec<i32>) {
909        //Function assumes valid board coordinates and valid move
910
911        let source_coords = source;
912        let target_coords = target; 
913        // the preceding 2 lines of code stem from laziness
914
915        let mut increment_halfmove_counter = true;
916        let piece = get_piece(self, &source_coords);
917
918        //Castling counter updates: Rook move => that side castling is disabled
919        if piece == 'R' && source_coords[1] == 0 {
920            self.castling_availability = self.castling_availability.replace(&'Q'.to_string(), "");
921        } else if piece == 'R' && source_coords[1] == 7 {
922            self.castling_availability = self.castling_availability.replace(&'K'.to_string(), "");
923        } else if piece == 'r' && source_coords[1] == 0 {
924            self.castling_availability = self.castling_availability.replace(&'q'.to_string(), "");
925        } else if piece == 'r' && source_coords[1] == 7 {
926            self.castling_availability = self.castling_availability.replace(&'k'.to_string(), "");
927        }
928
929        if piece == 'K' {
930            self.castling_availability = self.castling_availability.replace(&'Q'.to_string(), "");
931            self.castling_availability = self.castling_availability.replace(&'K'.to_string(), "");
932        } else if piece == 'k' {
933            self.castling_availability = self.castling_availability.replace(&'q'.to_string(), "");
934            self.castling_availability = self.castling_availability.replace(&'k'.to_string(), "");
935        }
936        if self.castling_availability.is_empty() {
937            self.castling_availability = "-".to_string();
938        }
939
940
941        if get_piece(self, &target_coords) != '*' { // target square isn't empty => Capture
942            increment_halfmove_counter = false;
943        }
944        // OR
945        if piece.eq_ignore_ascii_case(&'p')// the piece moved is a pawn
946        {
947            increment_halfmove_counter = false;
948            //Set en passant counter if it's a first turn advance
949            if target_coords[0] == source_coords[0] + 2 {
950                self.en_passant_square = get_algebraic_notation(vec![source_coords[0]+1, source_coords[1]]);
951            }
952            if target_coords[0] == source_coords[0] - 2 {
953                self.en_passant_square = get_algebraic_notation(vec![source_coords[0]-1, source_coords[1]]);
954            }
955        } else {
956            self.en_passant_square = "-".to_string();
957        }
958        if increment_halfmove_counter {
959            self.halfmove_counter += 1
960        } else {
961            self.halfmove_counter = 0
962        }
963        self.set_piece(&source_coords, '*');
964
965        if piece.eq_ignore_ascii_case(&'p') && (target_coords[0] == 0 || target_coords[0] == 7) { //Special case: Pawn promotion
966            if self.active_player == 'w' {
967                self.set_piece(&target_coords, self.promotion_selection.to_ascii_uppercase());
968            } else if self.active_player == 'b' {
969                self.set_piece(&target_coords, self.promotion_selection.to_ascii_lowercase());
970            }
971        } else {
972            self.set_piece(&target_coords, piece);
973        }
974        // Special case: En Passant
975        if self.en_passant_square != "-".to_string() {
976            if target_coords == get_board_coords(self.en_passant_square.clone()) {
977                match self.active_player {
978                    'w' => self.set_piece(&vec![target_coords[0]+1, target_coords[1]], '*'),
979                    'b' => self.set_piece(&vec![target_coords[0]-1, target_coords[1]], '*'),
980                    _ => panic!("ACTIVE PLAYER DOES NOT EXIST")
981                }
982            }
983        }
984
985        // Special case: Castling
986        if piece == 'K' && source_coords[1]-target_coords[1] == 2 {
987            self.set_piece(&target_coords, piece);
988            self.set_piece(&vec![7,0], '*');
989            self.set_piece(&vec![target_coords[0], target_coords[1]+1], 'R');
990        }
991        if piece == 'K' && source_coords[1]-target_coords[1] == -2 {
992            self.set_piece(&target_coords, piece);
993            self.set_piece(&vec![7,7], '*');
994            self.set_piece(&vec![target_coords[0], target_coords[1]-1], 'R');
995        }
996        if piece == 'k' && source_coords[1]-target_coords[1] == 2 {
997            self.set_piece(&target_coords, piece);
998            self.set_piece(&vec![0,0], '*');
999            self.set_piece(&vec![target_coords[0], target_coords[1]+1], 'R');
1000        }
1001        if piece == 'K' && source_coords[1]-target_coords[1] == -2 {
1002            self.set_piece(&target_coords, piece);
1003            self.set_piece(&vec![7,7], '*');
1004            self.set_piece(&vec![target_coords[0], target_coords[1]-1], 'R');
1005        }
1006    } // Moves a piece to a target square.

1009        self.board_state[coords[0] as usize][coords[1] as usize] = piece;
1010    } // changes the given board coordinate to the given piece.
